stay_in_alliance_third_international = {
    enable = {
        AND = {
            is_in_faction = yes
            any_allied_country = {
                tag = GER
                is_faction_leader = yes
            }
        }
    }
 
    abort = {
        NOT = { country_exists = GER }
    }

    ai_strategy = {
        type = alliance
        id = "GER"
        value = 500
    }
}
