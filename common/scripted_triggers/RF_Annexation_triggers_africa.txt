#Algeria - annexation_ALG_CONQ
can_release_algeria = {
	state = 459 #Algiers
	is_claim_or_core_of_ROOT = no
	ROOT = {
		# Do France release first
		OR = {
			NOT = { owns_state = 16 } #Ile de France
			has_country_flag = FRA_CONQ
		}
		NOT = {
			tag = FRA
			has_country_flag = ALG_CONQ
			country_exists = ALG
		}
	}
}
